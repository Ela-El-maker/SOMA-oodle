# INSTALL ORDER MATTERS on Windows + CUDA:
#
# Step 1 — PyTorch with CUDA (replace cu124 with your CUDA version):
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu124
#
# Step 2 — unsloth (installs optimised kernels for your GPU):
#   pip install "unsloth[colab-new] @ git+https://github.com/unslothai/unsloth.git"
#   # or stable release:
#   pip install unsloth
#
# Step 3 — everything else:
#   pip install -r requirements-training.txt
#
# Virtual env (recommended — keeps training deps separate from SOMA's Node runtime):
#   python -m venv .soma_venv
#   .soma_venv\Scripts\activate
#   <run steps 1-3 above>
#
